1. # git clone https://github.com/timo-wafula/btcscan.git
2. # cd btcscan
3. # pip install -r requirements.txt

# pip install python-dotenv
# pip install requests
# pip install bip-utils
# pip install pyfiglet

# python breadcracker.py
